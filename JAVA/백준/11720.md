```java
import java.util.Scanner;


public class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        // 입력값을 문자열로 저장후 배열에 각각의 문자로 변환
        String num1 = sc.next();
        char num2[] = num1.toCharArray();
        
        int sum=0;
        
        for(int i=0; i<N; i++){
            sum+=num2[i]-48;   // or -'0' (아스키코드 활용)
        }
        System.out.println(sum);
    }
}
```
