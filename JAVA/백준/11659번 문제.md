import java.util.Scanner;


public class Main {
    public static void main(String[] args){		
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt(); //함수로 입력데이터 저장
        int M = sc.nextInt();
        long A[] = new long[N+1];
        long S[] = new long[N+1];
        long sum=0;
        for(int i=1; i<=N; i++) {
        	//숫자 N개 입력받기
        	A[i] = sc.nextInt();
            //합배열
        	S[i] = S[i-1] + A[i];
        }
        
        for(int k=0; k<M; k++) {
        	//숫자 두개 입력받기
        	int i = sc.nextInt();
        	int j = sc.nextInt();
            sum = S[j]-S[i-1];
            System.out.println(sum);
        	}
        }
    }
